Python
import cv2
import numpy as np

# 1. تحويل منطقة التفاعل المركزية إلى HSV
hsv_roi = cv2.cvtColor(reaction_zone, cv2.COLOR_BGR2HSV)

# 2. حساب التدرج العمودي لقناة السطوع (V) وقناة التشبع (S) لتقصي خط التماس الحاد
sobel_y_sat = cv2.Sobel(hsv_roi[:,:,1], cv2.CV_64F, 0, 1, ksize=3)
max_gradient = np.max(np.abs(sobel_y_sat))

# 3. إذا تجاوز التدرج اللوني العتبة الحرجة (Threshold) ووصل قلب الشريط للبياض
if max_gradient > 150 and np.mean(hsv_roi[:,:,1]) < 10:
    # تصنيف فوري: تركيز حرج أكبر من 750 ppm
    ppm_status = "CRITICAL_HIGH_750"
